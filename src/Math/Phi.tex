\subsection{欧拉函数}
单独求欧拉函数phi(x)
    \begin{lstlisting}[language=c++]
int phi(int n) {
	int m = 1;
	for (int i = 2; n > 1; ++i) {
		if (n % i == 0) {
			m *= i - 1;
			n /= i;
			while (n % i == 0) {
				m *= i;
				n /= i;
			}
		}
	}
	return m;
}
    \end{lstlisting}
