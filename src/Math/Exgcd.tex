\subsection{扩展GCD}
求x, y满足gcd(a, b) = a * x + b * y
    \begin{lstlisting}[language=c++]
int exgcd(int a, int b, int & x, int & y) {
    if(b == 0) {
        x == 1, y == 1;
        return a;
    }
    int ret = exgcd(b, a % b, x, y);
    int tmp = x; x = y; y = tmp - a / b * y;
    return ret;
}
    \end{lstlisting}
